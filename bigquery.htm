Plik z danymi o wielkośći 0,86GB -

bigquery-public-data.open_images.images\

 

Dzięki połączeniu języka zapytań SQL oraz usługi BigQuery możemy bardzo szybko przeanalizować duże zbiory danych. W tym wypadku analizowałem dane z ogólnodostępnych danych ze zbioru GitHub Activity.

 

 

-- 1. Wybieram 5 pierwszych wierszy z tabeli images ze zbioru open_images:

 

SELECT * FROM `bigquery-public-data.open_images.images` LIMIT 5;
 


-- 2. Liczę liczbę wierszy w tabeli images:

 

SELECT COUNT(*) AS count_images FROM `bigquery-public-data.open_images.images`;
 


-- 3. Liczę unikalną liczbę wierszy w tabeli images w oparciu o kolumnę image_id:

 

SELECT COUNT(DISTINCT image_id) AS distinct_count FROM `bigquery-public-data.open_images.images`;
 


-- 4. Liczę unikalną liczbę wartości kolumny title:
 

SELECT COUNT(DISTINCT title) AS distinct_count_TITLE FROM `bigquery-public-data.open_images.images`;
 


-- 5. Pobieram do wyświetlenia kolumny: image_id, subset, original_url, title, original_size z tabeli images ze zbioru open_images:

 

SELECT
  image_id,
  subset,
  original_url,
  title,
  original_size
FROM
  `bigquery-public-data.open_images.images`
LIMIT 5;
 


-- 6. Wybieram 5 zdjęć z najmniejszym rozmiarem z tabeli images:
 

SELECT
  image_id,
  subset,
  original_url,
  title,
  original_size
FROM
  `bigquery-public-data.open_images.images`
ORDER BY
  original_size
LIMIT 5;
 


-- 7. Wybieram 5 zdjęć z największym rozmiarem z tabeli images:
 

SELECT
  image_id,
  subset,
  original_url,
  title,
  original_size
FROM
  `bigquery-public-data.open_images.images`
ORDER BY
  original_size DESC
LIMIT 5;
 


-- 8. Wybieram unikalne wartości dla kolumny subset:


SELECT DISTINCT subset AS distinct_subset FROM `bigquery-public-data.open_images.images`;
 


-- 9. Liczę częstość występowania każdego rodzaju zbioru w tabeli images:

 

SELECT
  subset,
  COUNT(subset) AS count_subset
FROM
  `bigquery-public-data.open_images.images`
GROUP BY
  subset;



-- 10. Liczę procentowo częstość występowania każdego rodzaju zbioru w tabeli images, wynik podaje z dokładnością do 4 miejsca po przecinku:

 

SELECT
  subset,
  ROUND(COUNT(subset) / (SELECT COUNT(subset) FROM `bigquery-public-data.open_images.images`), 4) AS share
FROM
  `bigquery-public-data.open_images.images`
GROUP BY
  subset;

 


-- 11. Wybieram 5 pierwszych wierszy z tabeli labels ze zbioru open_images:

 

SELECT * FROM `bigquery-public-data.open_images.labels` LIMIT 5;



-- 12. Łączę metodą INNER JOIN tabele images oraz labels, następnie wyświetlam kolumny image_id, subset, original_url, title, original_size, source, label_name, confidence oraz wyświetlam 5 pierwszych wierszy:

 

SELECT
  t1.image_id,
  t1.subset,
  t1.original_url,
  t1.title,
  t1.original_size,
  t2.source,
  t2.label_name,
  t2.confidence
FROM
  `bigquery-public-data.open_images.images` AS t1
INNER JOIN
  `bigquery-public-data.open_images.labels` AS t2
ON
  t1.image_id = t2.image_id
LIMIT 5;



-- 13. Wybieram 5 pierwszych wierszy z tabeli dict ze zbioru open_images:

 

SELECT * FROM `bigquery-public-data.open_images.dict` LIMIT 5;



-- 14. Łączę metodą INNER JOIN tabele images oraz dict. Następnie wyświetlam kolumny image_id, subset, original_url, title, original_size, source, label_name, confidence, label_display_name oraz wyświetlam 5 pierwszych wierszy:

 

SELECT
  t1.image_id,
  t1.subset,
  t1.original_url,
  t1.title,
  t1.original_size,
  t2.source,
  t2.label_name,
  t2.confidence,
  t3.label_display_name
FROM
  `bigquery-public-data.open_images.images` AS t1
INNER JOIN
  `bigquery-public-data.open_images.labels` AS t2
ON
  t1.image_id = t2.image_id
INNER JOIN
  `bigquery-public-data.open_images.dict` AS t3
ON
  t2.label_name = t3.label_name
LIMIT 5;
